파이썬으로 계산기를 만들어 봅시다. 간단한 사칙연산과 제곱, 절댓값, 제곱근, 삼각함수 등을 계산할 수 있습니다. 다음 예제를 보고 따라해 봅시다.
```
import math
a=input("어떤 것을 계산할지 입력하세요.")
b=int(input("1번째 수를 입력하세요."))
c=int(input("2번째 수를 입력하세요."))
if a=="나눗셈" or a=="/":
    if c==0:
        print("입력 유형이 잘못되었습니다.")
    else:
        print(b/c)
if a=="제곱근":
    if b<0:
        print("입력 유형이 잘못되었습니다.")
    else:
        print(math.sqrt(b))
result1={"+":b+c,"-":b-c,"*":b*c,"제곱":b**c,"덧셈":b+c,"뺄셈":b-c,"곱셈":b*c}
result2={"절댓값":abs(b),"sin":math.sin(math.radians(b)),"cos":math.cos(math.radians(b)),"tan":math.tan(math.radians(b))}
if a!="제곱근" and a!="나눗셈" and a!="/":
    if a in result1:
        print(result1[a])
    elif a in result2:
        print(result2[a])
    else:
        print("입력 유형이 잘못되었습니다.")

```
