파이썬으로 계산기를 만들어 봅시다. 간단한 사칙연산과 제곱, 절댓값, 제곱근, 삼각함수 등을 계산할 수 있습니다. 다음 예제를 보고 따라해 봅시다.
```
import math
a=input("어떤 것을 계산할지 입력하세요.")
if a=="+" or a=="덧셈" or a=="-" or a=="뺄셈" or a=="*" or a=="곱셈" or a=="/" or a=="나눗셈" or a=="제곱":
    b=int(input("1번째 수를 입력하세요."))
    c=int(input("2번째 수를 입력하세요."))
    if a=="+" or a=="덧셈":
        result=b+c
        print(result)
    elif a=="-" or a=="뺄셈":
        result=b-c
        print(result)
    elif a=="*" or a=="곱셈":
        result=c*b
        print(result)
    elif a=="/" or a=="나눗셈":
        result=b/c
        print(result)
    elif a=="제곱":
        result=b**c
        print(result)
elif a=="제곱근" or a=="sin" or a=="cos" or a=="tan" or a=="절댓값":
    b=int(input("수를 입력하세요."))
else:
    print("입력 유형이 잘못되었습니다.")
```
